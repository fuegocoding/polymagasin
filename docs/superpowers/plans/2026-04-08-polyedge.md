# PolyEdge Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a working Polymarket ↔ sportsbook arbitrage scanner (Pinnacle + Stake + Mise-o-jeu) that detects ≥5% edge opportunities, logs all signals to SQLite, and supports paper trading P&L tracking via a Typer CLI.

**Architecture:** Modular Python package `polyedge/` with isolated async fetchers per source, a rapidfuzz event matcher, a multiplicative-devig edge calculator, SQLite signal storage, and a Typer/Rich CLI. All fetchers run concurrently via `asyncio.gather`. Fetcher failures are isolated — other sources continue.

**Tech Stack:** Python 3.11+, httpx (async HTTP), typer (CLI), rich (table output), rapidfuzz (fuzzy matching), tomllib (config, stdlib), sqlite3 (DB, stdlib), pytest + pytest-asyncio + respx (testing)

from __future__ import annotations
import tomllib
from dataclasses import dataclass, field


@dataclass
class ScannerConfig:
    edge_threshold: float = 0.05
    scan_interval_minutes: int = 30
    stale_odds_minutes: int = 30
    bankroll: float = 500.0


@dataclass
class Config:
    scanner: ScannerConfig
    sports: list[str]
    sources: dict[str, bool]
    db_path: str
    stake_api_key: str = ""
    telegram_token: str = ""
    telegram_chat_id: str = ""


def load_config(path: str = "config.toml") -> Config:
    with open(path, "rb") as f:
        data = tomllib.load(f)

    raw_scanner = data.get("scanner", {})
    scanner = ScannerConfig(
        edge_threshold=raw_scanner.get("edge_threshold", 0.05),
        scan_interval_minutes=raw_scanner.get("scan_interval_minutes", 30),
        stale_odds_minutes=raw_scanner.get("stale_odds_minutes", 30),
        bankroll=raw_scanner.get("bankroll", 500.0),
    )

    return Config(
        scanner=scanner,
        sports=data.get("sports", {}).get("enabled", ["nba", "nhl", "mlb", "epl"]),
        sources=data.get("sources", {"pinnacle": True, "stake": True, "miseonjeu": True}),
        db_path=data.get("db", {}).get("path", "polyedge.db"),
        stake_api_key=data.get("keys", {}).get("stake_api_key", ""),
        telegram_token=data.get("keys", {}).get("telegram_token", ""),
        telegram_chat_id=data.get("keys", {}).get("telegram_chat_id", ""),
    )
```

    sport           TEXT NOT NULL,
    league          TEXT NOT NULL,
    team1           TEXT NOT NULL,

    suggested_size  REAL NOT NULL,
    sources_used    TEXT NOT NULL,
    status          TEXT NOT NULL DEFAULT 'pending',
    outcome_price   REAL,
    pnl             REAL
)
"""

_CREATE_SCAN_LOGS = """
CREATE TABLE IF NOT EXISTS scan_logs (
    id               INTEGER PRIMARY KEY AUTOINCREMENT,
    timestamp        TEXT NOT NULL,
    markets_scanned  INTEGER NOT NULL,
    signals_found    INTEGER NOT NULL,
    sources_active   TEXT NOT NULL,
    duration_ms      INTEGER NOT NULL
)
"""

    conn = sqlite3.connect(path)
    conn.row_factory = sqlite3.Row

from __future__ import annotations
import sqlite3
from datetime import datetime, timezone
from polyedge.models import Signal


def insert_signal(conn: sqlite3.Connection, signal: Signal) -> int:
    cur = conn.execute(
        """INSERT INTO signals
           (timestamp, sport, league, team1, team2, game_date,
            edge_pct, poly_price, poly_market_id, fair_value,
            kelly_fraction, suggested_size, sources_used, status)
           VALUES (?,?,?,?,?,?,?,?,?,?,?,?,?,?)""",
        (
            signal.timestamp.isoformat(),
            signal.sport,
            signal.league,
            signal.team1,
            signal.team2,
            signal.game_date.isoformat(),
            signal.edge_pct,
            signal.poly_price,
            signal.poly_market_id,
            signal.fair_value,
            signal.kelly_fraction,
            signal.suggested_size,
            signal.sources_used,
            signal.status,
        ),
    )
    conn.commit()
    return cur.lastrowid


def get_signal_by_id(conn: sqlite3.Connection, signal_id: int) -> Signal:
    row = conn.execute("SELECT * FROM signals WHERE id=?", (signal_id,)).fetchone()
    if row is None:
        raise ValueError(f"Signal {signal_id} not found")
    return _row_to_signal(row)


def get_signals(
    conn: sqlite3.Connection,
    sport: str | None = None,
    min_edge: float = 0.0,
    status: str | None = None,
) -> list[Signal]:
    query = "SELECT * FROM signals WHERE edge_pct >= ?"
    params: list = [min_edge]
    if sport:
        query += " AND sport = ?"
        params.append(sport)
    if status:
        query += " AND status = ?"
        params.append(status)
    query += " ORDER BY timestamp DESC"
    rows = conn.execute(query, params).fetchall()
    return [_row_to_signal(r) for r in rows]


def resolve_signal(
    conn: sqlite3.Connection,
    signal_id: int,
    status: str,
    outcome_price: float,
) -> None:
    """status: 'won' | 'lost' | 'push'"""
    sig = get_signal_by_id(conn, signal_id)
    if status == "won":
        pnl = sig.suggested_size * (1.0 / sig.poly_price - 1.0)
    elif status == "lost":
        pnl = -sig.suggested_size
    else:
        pnl = 0.0
    conn.execute(
        "UPDATE signals SET status=?, outcome_price=?, pnl=? WHERE id=?",
        (status, outcome_price, pnl, signal_id),
    )
    conn.commit()


def get_pnl_by_sport(conn: sqlite3.Connection) -> dict[str, float]:
    rows = conn.execute(
        "SELECT sport, SUM(pnl) as total FROM signals WHERE pnl IS NOT NULL GROUP BY sport"
    ).fetchall()
    return {r["sport"]: r["total"] for r in rows}


def log_scan(
    conn: sqlite3.Connection,
    markets_scanned: int,
    signals_found: int,
    sources_active: list[str],
    duration_ms: int,
) -> None:
    conn.execute(
        """INSERT INTO scan_logs
           (timestamp, markets_scanned, signals_found, sources_active, duration_ms)
           VALUES (?,?,?,?,?)""",
        (
            datetime.now(timezone.utc).isoformat(),
            markets_scanned,
            signals_found,
            ",".join(sources_active),
            duration_ms,
        ),
    )
    conn.commit()


def _row_to_signal(row: sqlite3.Row) -> Signal:
    return Signal(
        id=row["id"],
        timestamp=datetime.fromisoformat(row["timestamp"]),
        sport=row["sport"],
        league=row["league"],
        team1=row["team1"],
        team2=row["team2"],
        game_date=datetime.fromisoformat(row["game_date"]),
        edge_pct=row["edge_pct"],
        poly_price=row["poly_price"],
        poly_market_id=row["poly_market_id"],
        fair_value=row["fair_value"],
        kelly_fraction=row["kelly_fraction"],
        suggested_size=row["suggested_size"],
        sources_used=row["sources_used"],
        status=row["status"],
        outcome_price=row["outcome_price"],
        pnl=row["pnl"],
    )

import pytest
from polyedge.edge.calculator import devig, implied_prob, calculate_edge, EdgeResult

    # decimal 1.25 → 80%
    assert implied_prob(1.25) == pytest.approx(0.80)


def test_devig_fair_probabilities():
    # home 2.10, away 1.80 — sum of implied > 1.0 (overround)
    fair_home, fair_away = devig(2.10, 1.80)
    assert fair_home + fair_away == pytest.approx(1.0, abs=1e-9)
    # home is the underdog (higher odds), so fair_home < fair_away
    assert fair_home < fair_away


def test_devig_even_odds():
    # both 2.00 with no vig → both exactly 50%
    # with no vig: implied = 0.5 each, overround = 1.0

class EdgeResult:
    edge_pct: float
    fair_value: float


def implied_prob(decimal_odds: float) -> float:
    """Convert decimal odds to implied probability."""
    return 1.0 / decimal_odds


def devig(odds_home: float, odds_away: float) -> tuple[float, float]:
    """Remove vig using multiplicative method. Returns (fair_home, fair_away)."""
    raw_home = implied_prob(odds_home)
    raw_away = implied_prob(odds_away)
    overround = raw_home + raw_away
    return raw_home / overround, raw_away / overround


def calculate_edge(
    poly_price: float,
    fair_home: float,
    fair_away: float,
    team_is_home: bool,
) -> EdgeResult | None:
    """
    Returns EdgeResult if poly_price > fair_value, else None.
    team_is_home=True means YES token resolves if home team wins.
    """
    fair_value = fair_home if team_is_home else fair_away
    edge = poly_price - fair_value
    if edge <= 0:
        return None
    return EdgeResult(edge_pct=edge, fair_value=fair_value)


def average_fair_values(lines: list[tuple[float, float]]) -> tuple[float, float]:
    """Average fair (home, away) tuples from multiple deviggged sources."""
    avg_home = sum(h for h, _ in lines) / len(lines)
    avg_away = sum(a for _, a in lines) / len(lines)
    return avg_home, avg_away
```

def quarter_kelly_size(
    edge_pct: float,
    fair_value: float,
    bankroll: float,
) -> tuple[float, float]:
    """
    Returns (suggested_size_usd, kelly_fraction).
    Uses quarter-Kelly for safety. Caps at full bankroll.
    """
    if edge_pct <= 0:
        raise ValueError("edge_pct must be positive")

    # Full Kelly fraction = edge / (1 - fair_value)
    # (equivalent to b*p - q / b where b = 1/fair_value - 1)
    full_kelly = edge_pct / (1.0 - fair_value)
    quarter = full_kelly / 4.0

    size = min(quarter * bankroll, bankroll)
    size = max(size, 1.0)
    return round(size, 2), round(quarter, 6)
```

from __future__ import annotations
from dataclasses import dataclass
from datetime import timedelta

from rapidfuzz import fuzz

from polyedge.models import OddsLine, PolyMarket
from polyedge.matching.normalizer import normalize_team

_DATE_WINDOW = timedelta(hours=4)
_FUZZY_THRESHOLD = 85


@dataclass
class MatchResult:
    matched_lines: list[OddsLine]
    team_is_home: bool   # True if poly YES token = home team in odds


def find_matching_odds(poly: PolyMarket, lines: list[OddsLine]) -> MatchResult | None:
    """
    Find all OddsLines that match this PolyMarket event.
    Returns None if no matches found.
    """
    canon_yes = normalize_team(poly.team_yes, poly.sport)
    canon_no = normalize_team(poly.team_no, poly.sport)

    matched: list[OddsLine] = []
    team_is_home: bool | None = None

    for line in lines:
        if line.sport.lower() != poly.sport.lower():
            continue
        if abs(line.game_date - poly.game_date) > _DATE_WINDOW:
            continue

        canon1 = normalize_team(line.team1, line.sport)
        canon2 = normalize_team(line.team2, line.sport)

        home, is_home = _teams_match(canon_yes, canon_no, canon1, canon2)
        if home:
            matched.append(line)
            team_is_home = is_home

    if not matched:
        return None
    return MatchResult(matched_lines=matched, team_is_home=team_is_home)


def _teams_match(
    yes: str,
    no: str,
    team1: str,
    team2: str,
) -> tuple[bool, bool]:
    """
    Check if (yes, no) matches (team1, team2) in either order.
    Returns (matched, yes_is_home).
    Uses exact match first, then fuzzy fallback.
    """
    # Exact: yes=home, no=away
    if yes == team1 and no == team2:
        return True, True
    # Exact: yes=away, no=home
    if yes == team2 and no == team1:
        return True, False

    # Fuzzy fallback
    score_yes_home = min(
        fuzz.token_sort_ratio(yes, team1),
        fuzz.token_sort_ratio(no, team2),
    )
    score_yes_away = min(
        fuzz.token_sort_ratio(yes, team2),
        fuzz.token_sort_ratio(no, team1),
    )

    if score_yes_home >= _FUZZY_THRESHOLD and score_yes_home >= score_yes_away:
        return True, True
    if score_yes_away >= _FUZZY_THRESHOLD:
        return True, False

    return False, False
```

import asyncio
from abc import ABC, abstractmethod
import httpx
from polyedge.models import OddsLine


class BaseFetcher(ABC):
    def __init__(self, client: httpx.AsyncClient):
        self.client = client

    @abstractmethod
    async def fetch(self, sports: list[str]) -> list[OddsLine]:
        ...

    async def _get_json(self, url: str, **kwargs) -> list | dict:
        last_exc: Exception | None = None
        for attempt in range(3):
            try:
                resp = await self.client.get(url, timeout=15.0, **kwargs)
                resp.raise_for_status()
                return resp.json()
            except (httpx.HTTPStatusError, httpx.RequestError) as exc:
                last_exc = exc
                if attempt < 2:
                    await asyncio.sleep(2 ** attempt)
        raise last_exc

```python
from __future__ import annotations
import re
from datetime import datetime, timezone
import httpx

from polyedge.fetchers.base import BaseFetcher
from polyedge.models import OddsLine, PolyMarket
from polyedge.matching.normalizer import normalize_team

_BASE = "https://gamma-api.polymarket.com"
_SPORTS_SLUGS = {"nba", "nhl", "mlb", "epl", "soccer", "sports"}

# Patterns: "Will [A] beat [B] on ...", "Will [A] defeat [B]", "[A] vs [B]"
_WILL_BEAT = re.compile(
    r"Will (?:the )?(.+?) (?:beat|defeat|win (?:vs?\.?|against)) (?:the )?(.+?)(?:\s+on\b|\?|$)",
    re.IGNORECASE,
)
_VS = re.compile(r"^(.+?)\s+vs\.?\s+(.+?)(?:\s*[-–]|\?|$)", re.IGNORECASE)


class PolymarketFetcher:
    def __init__(self, client: httpx.AsyncClient):
        self.client = client

    async def fetch(self, sports: list[str]) -> list[PolyMarket]:
        data = await self._get_json(
            f"{_BASE}/markets",
            params={"active": "true", "closed": "false", "limit": 500},
        )
        results: list[PolyMarket] = []
        for item in data:
            market = self._parse_market(item, sports)
            if market:
                results.append(market)
        return results

    def _parse_market(self, item: dict, sports: list[str]) -> PolyMarket | None:
        tags = [t.get("slug", "").lower() for t in item.get("tags", [])]
        sport = next((t for t in tags if t in sports), None)
        if not sport:
            return None

        try:
            prices = item.get("outcomePrices", '["0.5","0.5"]')
            if isinstance(prices, str):
                import json as _json
                price_list = _json.loads(prices)
            else:
                price_list = prices
            price_yes = float(price_list[0])
        except (ValueError, IndexError, TypeError):
            return None

        tokens = item.get("tokens", [])
        token_yes = next((t["token_id"] for t in tokens if t.get("outcome") == "Yes"), None)
        if not token_yes:
            return None

        question = item.get("question", "")
        team_yes, team_no = self._extract_teams(question, sport)
        if not team_yes or not team_no:
            return None

        try:
            game_date = datetime.fromisoformat(
                item["startDate"].replace("Z", "+00:00")
            )
        except (KeyError, ValueError):
            return None

        slug = item.get("slug", item.get("id", ""))
        return PolyMarket(
            market_id=item["id"],
            question=question,
            token_id_yes=token_yes,
            price_yes=price_yes,
            sport=sport,
            team_yes=normalize_team(team_yes, sport),
            team_no=normalize_team(team_no, sport),
            game_date=game_date,
            url=f"https://polymarket.com/event/{slug}",
        )

    def _extract_teams(self, question: str, sport: str) -> tuple[str | None, str | None]:
        m = _WILL_BEAT.search(question)
        if m:
            return m.group(1).strip(), m.group(2).strip()
        m = _VS.search(question)
        if m:
            return m.group(1).strip(), m.group(2).strip()
        return None, None

    async def _get_json(self, url: str, **kwargs) -> list | dict:
        import asyncio
        last_exc: Exception | None = None
        for attempt in range(3):
            try:
                resp = await self.client.get(url, timeout=15.0, **kwargs)
                resp.raise_for_status()
                return resp.json()
            except Exception as exc:
                last_exc = exc
                if attempt < 2:
                    await asyncio.sleep(2 ** attempt)
        raise last_exc
```

from __future__ import annotations
import asyncio
from datetime import datetime
import httpx

from polyedge.fetchers.base import BaseFetcher
from polyedge.models import OddsLine
from polyedge.matching.normalizer import normalize_team

_BASE = "https://www.pinnacle.com/api/v3"

# Pinnacle league IDs for supported sports
_LEAGUE_IDS: dict[str, list[int]] = {
    "nba": [487],
    "nhl": [1456],
    "mlb": [246],
    "epl": [1980],
}


class PinnacleFetcher(BaseFetcher):
    async def fetch(self, sports: list[str]) -> list[OddsLine]:
        league_ids: list[int] = []
        sport_by_league: dict[int, str] = {}
        for sport in sports:
            ids = _LEAGUE_IDS.get(sport.lower(), [])
            for lid in ids:
                league_ids.append(lid)
                sport_by_league[lid] = sport.lower()

        if not league_ids:
            return []

        data = await self._get_json(
            f"{_BASE}/matchups",
            params={
                "leagueIds": ",".join(str(i) for i in league_ids),
                "withSpecials": "false",
                "brandId": "1",
            },
        )

        results: list[OddsLine] = []
        for matchup in data.get("matchups", []):
            line = self._parse_matchup(matchup, sport_by_league)
            if line:
                results.append(line)
        return results

    def _parse_matchup(
        self, matchup: dict, sport_by_league: dict[int, str]
    ) -> OddsLine | None:
        league_id = matchup.get("league", {}).get("id")
        sport = sport_by_league.get(league_id)
        if not sport:
            return None

        # Find period 0 (full game) moneyline
        ml = None
        for period in matchup.get("periods", []):
            if period.get("number") == 0 and "moneyline" in period:
                ml = period["moneyline"]
                break
        if not ml:
            return None

        participants = matchup.get("participants", [])
        home = next((p["name"] for p in participants if p.get("alignment") == "home"), None)
        away = next((p["name"] for p in participants if p.get("alignment") == "away"), None)
        if not home or not away:
            return None

        try:
            game_date = datetime.fromisoformat(
                matchup["startTime"].replace("Z", "+00:00")
            )
        except (KeyError, ValueError):
            return None

        league_name = matchup.get("league", {}).get("name", sport.upper())

        return OddsLine(
            source="pinnacle",
            sport=sport,
            league=league_name,
            team1=normalize_team(home, sport),
            team2=normalize_team(away, sport),
            game_date=game_date,
            odds_home=float(ml["home"]),
            odds_away=float(ml["away"]),
            fetched_at=datetime.now(datetime.now().astimezone().tzinfo),
        )

```python
from __future__ import annotations
import asyncio
from datetime import datetime, timezone
import httpx

from polyedge.fetchers.base import BaseFetcher
from polyedge.models import OddsLine
from polyedge.matching.normalizer import normalize_team

_URL = "https://stake.com/_api/graphql"

# Stake sport slugs mapped to our internal sport keys
_SPORT_SLUGS: dict[str, str] = {
    "basketball_nba": "nba",
    "ice_hockey_nhl": "nhl",
    "baseball_mlb": "mlb",
    "soccer_england_premier_league": "epl",
}

_QUERY = """
query SportsbookEventList($sportSlug: String!, $limit: Int) {
  sportsbookEventList(sportSlug: $sportSlug, limit: $limit, status: UPCOMING) {
    id
    name
    startTime
    sport { slug }
    markets(name: "Match Winner") {
      name
      outcomes {
        name
        price
      }
    }
  }
}
"""


class StakeFetcher(BaseFetcher):
    async def fetch(self, sports: list[str]) -> list[OddsLine]:
        # Build reverse map: our sport key → Stake slug
        slug_map = {v: k for k, v in _SPORT_SLUGS.items()}
        tasks = [
            self._fetch_sport(sport, slug_map[sport])
            for sport in sports
            if sport in slug_map
        ]
        results: list[OddsLine] = []
        for group in await asyncio.gather(*tasks, return_exceptions=True):
            if isinstance(group, Exception):
                print(f"[stake] fetch error: {group}")
                continue
            results.extend(group)
        return results

    async def _fetch_sport(self, sport: str, slug: str) -> list[OddsLine]:
        last_exc: Exception | None = None
        for attempt in range(3):
            try:
                resp = await self.client.post(
                    _URL,
                    json={
                        "query": _QUERY,
                        "variables": {"sportSlug": slug, "limit": 200},
                    },
                    timeout=15.0,
                    headers={"Content-Type": "application/json"},
                )
                resp.raise_for_status()
                data = resp.json()
                events = data.get("data", {}).get("sportsbookEventList", [])
                return [
                    line
                    for event in events
                    for line in [self._parse_event(event, sport)]
                    if line
                ]
            except Exception as exc:
                last_exc = exc
                if attempt < 2:
                    await asyncio.sleep(2 ** attempt)
        raise last_exc

    def _parse_event(self, event: dict, sport: str) -> OddsLine | None:
        markets = event.get("markets", [])
        winner_market = next(
            (m for m in markets if "winner" in m.get("name", "").lower()),
            None,
        )
        if not winner_market:
            return None

        outcomes = winner_market.get("outcomes", [])
        if len(outcomes) < 2:
            return None

        home_outcome = outcomes[0]
        away_outcome = outcomes[1]

        # Parse team names from event name "Team A vs Team B"
        name = event.get("name", "")
        parts = [p.strip() for p in name.split(" vs ")]
        if len(parts) == 2:
            raw_home, raw_away = parts[0], parts[1]
        else:
            raw_home = home_outcome["name"]
            raw_away = away_outcome["name"]

        try:
            game_date = datetime.fromisoformat(
                event["startTime"].replace("Z", "+00:00")
            )
        except (KeyError, ValueError):
            return None

        return OddsLine(
            source="stake",
            sport=sport,
            league=sport.upper(),
            team1=normalize_team(raw_home, sport),
            team2=normalize_team(raw_away, sport),
            game_date=game_date,
            odds_home=float(home_outcome["price"]),
            odds_away=float(away_outcome["price"]),
            fetched_at=datetime.now(timezone.utc),
        )

    "mlb": "BASEBALL/MLB",
    "epl": "FOOTBALL/ENGLAND/PREMIER_LEAGUE",
}


class MiseonjeuFetcher(BaseFetcher):
    async def fetch(self, sports: list[str]) -> list[OddsLine]:
        tasks = [
            self._fetch_sport(sport)
            for sport in sports
            if sport in _KAMBI_SPORTS
        ]
        results: list[OddsLine] = []
        for group in await asyncio.gather(*tasks, return_exceptions=True):
            if isinstance(group, Exception):
                print(f"[miseonjeu] fetch error: {group}")
                continue
            results.extend(group)
        return results

    async def _fetch_sport(self, sport: str) -> list[OddsLine]:
        sport_path = _KAMBI_SPORTS[sport]
        url = f"{_BASE}/listView/{sport_path.lower().replace('/', '_')}.json"
        last_exc: Exception | None = None
        for attempt in range(3):
            try:
                resp = await self.client.get(
                    url,
                    params={"lang": "fr_CA", "market": "CA", "client_id": "2", "channel_id": "1"},
                    timeout=15.0,
                )
                resp.raise_for_status()
                return self._parse_response(resp.json(), sport)
            except Exception as exc:
                last_exc = exc
                if attempt < 2:
                    await asyncio.sleep(2 ** attempt)
        print(f"[miseonjeu] failed for {sport} after 3 attempts: {last_exc}")
        return []

    def _parse_response(self, data: dict, sport: str) -> list[OddsLine]:
        results: list[OddsLine] = []
        for event in data.get("events", []):
            line = self._parse_event(event, sport)
            if line:
                results.append(line)
        return results

    def _parse_event(self, event: dict, sport: str) -> OddsLine | None:
        try:
            event_data = event.get("event", event)
            name = event_data.get("name", "")
            start = event_data.get("start", "")
            game_date = datetime.fromisoformat(start.replace("Z", "+00:00"))

            # Kambi event name: "Home Team - Away Team"
            if " - " in name:
                home_raw, away_raw = name.split(" - ", 1)
            else:
                return None

            # Find 1X2 or moneyline market
            bet_offers = event_data.get("betOffers", [])
            ml_offer = next(
                (
                    b for b in bet_offers
                    if b.get("betOfferType", {}).get("name", "") in ("Match", "1X2", "Moneyline", "To Win")
                ),
                None,
            )
            if not ml_offer:
                return None

            outcomes = ml_offer.get("outcomes", [])
            home_out = next((o for o in outcomes if o.get("label") in ("1", "Home", home_raw)), None)
            away_out = next((o for o in outcomes if o.get("label") in ("2", "Away", away_raw)), None)
            if not home_out or not away_out:
                return None

            # Kambi odds are in milliodds (e.g., 1850 = decimal 1.85)
            odds_home = home_out["odds"] / 1000.0
            odds_away = away_out["odds"] / 1000.0

            return OddsLine(
                source="miseonjeu",
                sport=sport,
                league=sport.upper(),
                team1=normalize_team(home_raw.strip(), sport),
                team2=normalize_team(away_raw.strip(), sport),
                game_date=game_date,
                odds_home=odds_home,
                odds_away=odds_away,
                fetched_at=datetime.now(timezone.utc),
            )
        except Exception:
            return None
```

```python
from __future__ import annotations
import sqlite3
from datetime import datetime, timezone, timedelta

from polyedge.config import Config
from polyedge.models import OddsLine, PolyMarket, Signal
from polyedge.matching.matcher import find_matching_odds
from polyedge.edge.calculator import devig, calculate_edge, average_fair_values
from polyedge.edge.kelly import quarter_kelly_size
from polyedge.db.signals import insert_signal


def run_scan(
    poly_markets: list[PolyMarket],
    odds_lines: list[OddsLine],
    config: Config,
    conn: sqlite3.Connection,
) -> list[Signal]:
    """
    Pure synchronous signal generation — fetching is done by caller.
    Matches markets, calculates edges, saves signals to DB.
    """
    threshold = config.scanner.edge_threshold
    bankroll = config.scanner.bankroll
    stale_cutoff = timedelta(minutes=config.scanner.stale_odds_minutes)
    now = datetime.now(timezone.utc)

    # Filter stale odds
    fresh_lines = [
        l for l in odds_lines
        if (now - l.fetched_at.replace(tzinfo=timezone.utc) if l.fetched_at.tzinfo is None
            else now - l.fetched_at) <= stale_cutoff
    ]

    signals: list[Signal] = []

    for market in poly_markets:
        match = find_matching_odds(market, fresh_lines)
        if not match or len(match.matched_lines) < 1:
            continue

        # Devig each source and average fair values
        fair_pairs = [devig(l.odds_home, l.odds_away) for l in match.matched_lines]
        avg_fair_home, avg_fair_away = average_fair_values(fair_pairs)

        edge_result = calculate_edge(
            poly_price=market.price_yes,
            fair_home=avg_fair_home,
            fair_away=avg_fair_away,
            team_is_home=match.team_is_home,
        )

        if not edge_result or edge_result.edge_pct < threshold:
            continue

        size, fraction = quarter_kelly_size(
            edge_pct=edge_result.edge_pct,
            fair_value=edge_result.fair_value,
            bankroll=bankroll,
        )

        sources = ",".join(sorted({l.source for l in match.matched_lines}))
        league = match.matched_lines[0].league

        signal = Signal(
            timestamp=now,
            sport=market.sport,
            league=league,
            team1=market.team_yes if match.team_is_home else market.team_no,
            team2=market.team_no if match.team_is_home else market.team_yes,
            game_date=market.game_date,
            edge_pct=edge_result.edge_pct,
            poly_price=market.price_yes,
            poly_market_id=market.market_id,
            fair_value=edge_result.fair_value,
            kelly_fraction=fraction,
            suggested_size=size,
            sources_used=sources,
        )

        insert_signal(conn, signal)
        signals.append(signal)

    return signals
```

from __future__ import annotations
from datetime import datetime, timezone
from rich.console import Console
from rich.table import Table
from rich import box
from polyedge.models import Signal

console = Console()


def print_signals_table(signals: list[Signal], title: str = "PolyEdge Opportunities") -> None:
    if not signals:
        console.print(f"[yellow]No signals found above threshold.[/yellow]")
        return

    table = Table(
        title=title,
        box=box.ROUNDED,
        show_header=True,
        header_style="bold cyan",
    )
    table.add_column("Sport", style="dim", width=6)
    table.add_column("Matchup", min_width=28)
    table.add_column("Game Date", width=16)
    table.add_column("Edge %", justify="right", style="bold green")
    table.add_column("Poly Price", justify="right")
    table.add_column("Fair Value", justify="right")
    table.add_column("Sources", style="dim")
    table.add_column("Size ($)", justify="right", style="bold yellow")

    for sig in sorted(signals, key=lambda s: s.edge_pct, reverse=True):
        matchup = f"{sig.team1} vs {sig.team2}"
        game_str = sig.game_date.strftime("%b %d %H:%M")
        table.add_row(
            sig.sport.upper(),
            matchup,
            game_str,
            f"{sig.edge_pct * 100:.1f}%",
            f"{sig.poly_price:.3f}",
            f"{sig.fair_value:.3f}",
            sig.sources_used,
            f"${sig.suggested_size:.2f}",
        )

    console.print(table)


def print_pnl_table(pnl_by_sport: dict[str, float], total: float) -> None:
    table = Table(title="P&L by Sport", box=box.SIMPLE)
    table.add_column("Sport", style="bold")
    table.add_column("P&L ($)", justify="right")

    for sport, amount in sorted(pnl_by_sport.items()):
        color = "green" if amount >= 0 else "red"
        table.add_row(sport.upper(), f"[{color}]{amount:+.2f}[/{color}]")

    table.add_row("─" * 8, "─" * 10)
    color = "green" if total >= 0 else "red"
    table.add_row("[bold]TOTAL[/bold]", f"[bold {color}]{total:+.2f}[/bold {color}]")
    console.print(table)


def print_scan_summary(signals_found: int, markets_scanned: int, sources: list[str], duration_ms: int) -> None:
    console.print(
        f"[dim]Scanned {markets_scanned} markets · "
        f"{signals_found} signal(s) found · "
        f"Sources: {', '.join(sources)} · "
        f"{duration_ms}ms[/dim]"
    )

from __future__ import annotations
import asyncio
import time
from pathlib import Path
from datetime import datetime, timezone
from typing import Optional

import httpx
import typer
from rich.console import Console

from polyedge.config import load_config
from polyedge.db.schema import init_db
from polyedge.db.signals import get_signals, get_pnl_by_sport, resolve_signal, get_signal_by_id
from polyedge.fetchers.polymarket import PolymarketFetcher
from polyedge.fetchers.pinnacle import PinnacleFetcher
from polyedge.fetchers.stake import StakeFetcher
from polyedge.fetchers.miseonjeu import MiseonjeuFetcher
from polyedge.scanner import run_scan
from polyedge.cli.display import print_signals_table, print_pnl_table, print_scan_summary

app = typer.Typer(help="PolyEdge — Polymarket arbitrage scanner")
console = Console()


def _load(config_path: str = "config.toml"):
    cfg = load_config(config_path)
    conn = init_db(cfg.db_path)
    return cfg, conn


@app.command()
def scan(config: str = typer.Option("config.toml", help="Path to config.toml")):
    """Run one scan cycle and display opportunities."""
    asyncio.run(_do_scan(config))


@app.command()
def watch(config: str = typer.Option("config.toml", help="Path to config.toml")):
    """Continuously scan on the configured interval. Ctrl+C to stop."""
    cfg = load_config(config)
    interval = cfg.scanner.scan_interval_minutes * 60
    console.print(f"[cyan]Watching — scanning every {cfg.scanner.scan_interval_minutes} minutes. Ctrl+C to stop.[/cyan]")
    try:
        while True:
            asyncio.run(_do_scan(config))
            console.print(f"[dim]Next scan in {cfg.scanner.scan_interval_minutes} minutes…[/dim]")
            time.sleep(interval)
    except KeyboardInterrupt:
        console.print("\n[yellow]Stopped.[/yellow]")


@app.command()
def signals(
    sport: Optional[str] = typer.Option(None, help="Filter by sport (nba/nhl/mlb/epl)"),
    min_edge: float = typer.Option(0.0, help="Minimum edge % (e.g. 0.05 for 5%)"),
    status: Optional[str] = typer.Option(None, help="Filter by status: pending/won/lost/push"),
    config: str = typer.Option("config.toml"),
):
    """List signals from the database."""
    cfg, conn = _load(config)
    rows = get_signals(conn, sport=sport, min_edge=min_edge, status=status)
    if not rows:
        console.print("[yellow]No signals found.[/yellow]")
        return
    print_signals_table(rows, title=f"Signals (n={len(rows)})")


@app.command()
def pnl(config: str = typer.Option("config.toml")):
    """Show P&L summary by sport."""
    _, conn = _load(config)
    by_sport = get_pnl_by_sport(conn)
    if not by_sport:
        console.print("[yellow]No resolved signals yet.[/yellow]")
        return
    total = sum(by_sport.values())
    print_pnl_table(by_sport, total)


@app.command()
def resolve(
    signal_id: int = typer.Argument(help="Signal ID to resolve"),
    outcome: str = typer.Argument(help="Outcome: won / lost / push"),
    outcome_price: float = typer.Option(0.0, help="Polymarket resolution price (1.0=yes, 0.0=no)"),
    config: str = typer.Option("config.toml"),
):
    """Mark a signal as won/lost/push and record P&L."""
    if outcome not in ("won", "lost", "push"):
        console.print("[red]outcome must be: won / lost / push[/red]")
        raise typer.Exit(1)
    _, conn = _load(config)
    resolve_signal(conn, signal_id, outcome, outcome_price)
    sig = get_signal_by_id(conn, signal_id)
    color = "green" if sig.pnl >= 0 else "red"
    console.print(f"[bold]Signal {signal_id}[/bold] → {outcome} | P&L: [{color}]{sig.pnl:+.2f}[/{color}]")


async def _do_scan(config_path: str) -> None:
    cfg, conn = _load(config_path)
    t0 = time.monotonic()

    active_sources = [k for k, v in cfg.sources.items() if v]

    async with httpx.AsyncClient(headers={"User-Agent": "PolyEdge/1.0"}) as client:
        poly_fetcher = PolymarketFetcher(client)
        fetcher_map = {
            "pinnacle": PinnacleFetcher(client),
            "stake": StakeFetcher(client),
            "miseonjeu": MiseonjeuFetcher(client),
        }

        # Fetch Polymarket + all enabled sportsbooks concurrently
        poly_task = asyncio.create_task(poly_fetcher.fetch(cfg.sports))
        sb_tasks = {
            name: asyncio.create_task(fetcher.fetch(cfg.sports))
            for name, fetcher in fetcher_map.items()
            if cfg.sources.get(name, False)
        }

        poly_markets = await poly_task

        odds_lines = []
        for name, task in sb_tasks.items():
            try:
                lines = await task
                odds_lines.extend(lines)
            except Exception as e:
                console.print(f"[red][{name}] error: {e}[/red]")

    signals = await run_scan(
        poly_markets=poly_markets,
        odds_lines=odds_lines,
        config=cfg,
        conn=conn,
    )

    duration_ms = int((time.monotonic() - t0) * 1000)

    print_signals_table(signals)
    print_scan_summary(
        signals_found=len(signals),
        markets_scanned=len(poly_markets),
        sources=active_sources,
        duration_ms=duration_ms,
    )

    from polyedge.db.signals import log_scan
    log_scan(conn, len(poly_markets), len(signals), active_sources, duration_ms)
